## Task 2: decode.rs — data types and big-endian decoders

**Files:**
- Create: `crates/plugin-s7/src/s7/decode.rs`

S7 is big-endian. Each `DType` knows its byte length and how to turn raw bytes into `f64`.
`bool` is special-cased (read via the bit path, not bytes).

- [ ] **Step 1: Write the implementation:**

```rust
//! S7 data types: parse the selection `type`, know each type's byte length, and
//! decode big-endian PLC bytes into the f64 the warehouse stores.

/// A readable S7 scalar type. `Bool` is read via the bit path; the rest read `len()` bytes.
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum DType {
    Bool,
    SInt,   // i8
    USInt,  // u8  (alias: byte)
    Int,    // i16
    UInt,   // u16 (alias: word)
    DInt,   // i32
    UDInt,  // u32 (alias: dword)
    Real,   // f32
    LReal,  // f64
}

impl DType {
    /// Parse the selection `type` string (case-insensitive), accepting TIA/IEC aliases.
    pub fn parse(s: &str) -> Result<DType, String> {
        Ok(match s.to_ascii_lowercase().as_str() {
            "bool" | "bit" => DType::Bool,
            "sint" => DType::SInt,
            "usint" | "byte" => DType::USInt,
            "int" => DType::Int,
            "uint" | "word" => DType::UInt,
            "dint" => DType::DInt,
            "udint" | "dword" => DType::UDInt,
            "real" | "float" => DType::Real,
            "lreal" | "double" => DType::LReal,
            other => return Err(format!("unknown s7 type {other:?} (expected bool/sint/usint/int/uint/dint/udint/real/lreal or aliases byte/word/dword)")),
        })
    }

    /// Number of bytes to read for a byte-addressed read. `Bool` reads via the bit path → 0.
    pub fn byte_len(self) -> usize {
        match self {
            DType::Bool => 0,
            DType::SInt | DType::USInt => 1,
            DType::Int | DType::UInt => 2,
            DType::DInt | DType::UDInt | DType::Real => 4,
            DType::LReal => 8,
        }
    }

    /// Decode big-endian `buf` (length must be `byte_len()`) into f64. Not used for `Bool`.
    pub fn decode(self, buf: &[u8]) -> Result<f64, String> {
        let need = self.byte_len();
        if buf.len() < need {
            return Err(format!("s7 decode: need {need} bytes, got {}", buf.len()));
        }
        let b = buf;
        Ok(match self {
            DType::Bool => return Err("decode() called on Bool — use the bit path".into()),
            DType::SInt => (b[0] as i8) as f64,
            DType::USInt => b[0] as f64,
            DType::Int => i16::from_be_bytes([b[0], b[1]]) as f64,
            DType::UInt => u16::from_be_bytes([b[0], b[1]]) as f64,
            DType::DInt => i32::from_be_bytes([b[0], b[1], b[2], b[3]]) as f64,
            DType::UDInt => u32::from_be_bytes([b[0], b[1], b[2], b[3]]) as f64,
            DType::Real => f32::from_be_bytes([b[0], b[1], b[2], b[3]]) as f64,
            DType::LReal => f64::from_be_bytes([b[0], b[1], b[2], b[3], b[4], b[5], b[6], b[7]]),
        })
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn parses_aliases_case_insensitive() {
        assert_eq!(DType::parse("REAL").unwrap(), DType::Real);
        assert_eq!(DType::parse("word").unwrap(), DType::UInt);
        assert_eq!(DType::parse("DWord").unwrap(), DType::UDInt);
        assert_eq!(DType::parse("byte").unwrap(), DType::USInt);
        assert!(DType::parse("string").is_err());
    }

    #[test]
    fn byte_lengths() {
        assert_eq!(DType::Bool.byte_len(), 0);
        assert_eq!(DType::Int.byte_len(), 2);
        assert_eq!(DType::Real.byte_len(), 4);
        assert_eq!(DType::LReal.byte_len(), 8);
    }

    #[test]
    fn decodes_big_endian() {
        // REAL 3.14 (0x4048F5C3 big-endian)
        assert!((DType::Real.decode(&[0x40, 0x48, 0xF5, 0xC3]).unwrap() - 3.14).abs() < 1e-5);
        // INT -1 = 0xFFFF
        assert_eq!(DType::Int.decode(&[0xFF, 0xFF]).unwrap(), -1.0);
        // UINT 0xFFFF = 65535
        assert_eq!(DType::UInt.decode(&[0xFF, 0xFF]).unwrap(), 65535.0);
        // DINT -2 = 0xFFFFFFFE
        assert_eq!(DType::DInt.decode(&[0xFF, 0xFF, 0xFF, 0xFE]).unwrap(), -2.0);
        // SINT -1
        assert_eq!(DType::SInt.decode(&[0xFF]).unwrap(), -1.0);
        // LREAL 1.0 (0x3FF0000000000000)
        assert_eq!(DType::LReal.decode(&[0x3F, 0xF0, 0, 0, 0, 0, 0, 0]).unwrap(), 1.0);
    }

        assert!(DType::Real.decode(&[0x40, 0x48]).is_err());
```

```rust
use crate::s7::client::{S7_AREA_DB, S7_AREA_MK, S7_AREA_PE, S7_AREA_PA};
use crate::s7::decode::DType;
use serde_json::Value;

/// Resolve (rack, slot) from connector config: explicit rack+slot win; else `family`; else 0/1.
fn resolve_rack_slot(config: &Value) -> Result<(u16, u16), String> {
    let rack = config.get("rack").and_then(Value::as_u64);
    let slot = config.get("slot").and_then(Value::as_u64);
    if let (Some(r), Some(s)) = (rack, slot) {
        return Ok((r as u16, s as u16));
    }
    if let Some(fam) = config.get("family").and_then(Value::as_str) {
        let (r, s) = match fam.to_ascii_lowercase().as_str() {
            "s7-1500" | "s71500" | "s7-1200" | "s71200" => (0, 1),
            "s7-300" | "s7300" => (0, 2),
            "s7-400" | "s7400" => return Err(
                "family s7-400 has no fixed rack/slot — set rack and slot explicitly".into()),
            other => return Err(format!("unknown family {other:?} (s7-1500/s7-1200/s7-300, or set rack+slot)")),
        };
        // explicit rack or slot still override the family default if present
        return Ok((rack.map(|v| v as u16).unwrap_or(r), slot.map(|v| v as u16).unwrap_or(s)));
    }
    // default: S7-1200/1500 integrated PN
    Ok((rack.map(|v| v as u16).unwrap_or(0), slot.map(|v| v as u16).unwrap_or(1)))
}

struct Selection {
    area: u8,
    db: u16,
    byte: u16,
    bit: u8,
    dtype: DType,
}

fn parse_area(s: &str) -> Result<u8, String> {
    Ok(match s.to_ascii_lowercase().as_str() {
        "db" | "datablock" => S7_AREA_DB,
        "m" | "mk" | "merker" | "flag" => S7_AREA_MK,
        "i" | "e" | "input" | "pe" => S7_AREA_PE,
        "q" | "a" | "output" | "pa" => S7_AREA_PA,
        other => return Err(format!("unknown area {other:?} (db/m/i/q)")),
    })
}

fn parse_selection(sel: &Value) -> Result<Selection, String> {
    let area = match sel.get("area").and_then(Value::as_str) {
        Some(a) => parse_area(a)?,
        None => S7_AREA_DB,
    };
    let dtype = DType::parse(sel.get("type").and_then(Value::as_str).unwrap_or("real"))?;
    let db = sel.get("db").and_then(Value::as_u64).unwrap_or(0) as u16;
    if area == S7_AREA_DB && db == 0 && sel.get("db").is_none() {
        return Err("selection.db is required for a DB read".into());
    }
    let byte = sel.get("byte").and_then(Value::as_u64).unwrap_or(0) as u16;
    let bit = sel.get("bit").and_then(Value::as_u64).unwrap_or(0) as u8;
    if dtype == DType::Bool && bit > 7 {
        return Err(format!("selection.bit must be 0..7 for bool, got {bit}"));
    }
    Ok(Selection { area, db, byte, bit, dtype })
}

        assert_eq!(resolve_rack_slot(&json!({"rack":0,"slot":2})).unwrap(), (0, 2));
```

```rust
// Lymon connector plugin: Siemens S7 (S7-300/400/1200/1500 over ISO-on-TCP),
// built on lymon-collector-sdk and a vendored rust7 client (src/s7/client.rs).
//
// config:    { "host": "10.0.0.5", "family": "s7-1500" }   // or explicit "rack"/"slot"
// selection: { "area": "db", "db": 1, "byte": 0, "bit": 0, "type": "real" }
//            area = db|m|i|q ; type = bool/sint/usint/int/uint/dint/udint/real/lreal (+aliases)

mod s7;

use lymon_collector_sdk::{run, Collector, Discovery, Node, ReadRequest, Sample};
use s7::client::{S7Client, S7_AREA_DB, S7_WL_BYTE};
use s7::decode::DType;

// (resolve_rack_slot, Selection, parse_area, parse_selection — exactly as Task 3, here.)

/// A live S7 session for one (host,rack,slot), kept across polls.
struct Conn {
    key: (String, u16, u16),
    client: S7Client,
}

struct S7Connector {
    conn: Option<Conn>,
}

impl S7Connector {
    fn new() -> Self {
        Self { conn: None }
    }

    /// Get a connected client for (host,rack,slot), (re)connecting on miss/key-change.
    fn client_for(&mut self, host: &str, rack: u16, slot: u16) -> Result<&mut S7Client, String> {
        let key = (host.to_string(), rack, slot);
        let need = self.conn.as_ref().map(|c| c.key != key).unwrap_or(true);
        if need {
            let mut client = S7Client::new();
            // Bounded timeouts so a poll can't hang silently (OPC-UA lesson, agent #11).
            let _ = client.set_timeout(5000, 2000, 2000);
            client
                .connect_rack_slot(host, rack, slot)
                .map_err(|e| format!("s7 connect {host} rack{rack}/slot{slot}: {e}"))?;
            self.conn = Some(Conn { key, client });
        }
        Ok(&mut self.conn.as_mut().unwrap().client)
    }

    /// Drop the cached session so the next poll reconnects cleanly.
    fn reset(&mut self) {
        self.conn = None;
    }
}

impl Collector for S7Connector {
    fn read(&mut self, req: &ReadRequest) -> Result<Vec<Sample>, String> {
        let host = req.config_str("host").ok_or("config.host is required")?.to_string();
        let (rack, slot) = resolve_rack_slot(&req.config)?;
        let sel = parse_selection(&req.selection)?;
        let var_id = req.variable_id().unwrap_or("s7.value").to_string();

        let value = {
            let client = self.client_for(&host, rack, slot)?;
            let r = read_value(client, &sel);
            r
        };
        match value {
            Ok(v) => Ok(vec![Sample::new(var_id, v)]),
            Err(e) => {
                // Any wire error → drop the session so the next poll reconnects.
                self.reset();
                Err(map_s7_err(&e, &sel))
            }
        }
    }

    fn discover(&mut self, req: &ReadRequest) -> Result<Discovery, String> {
        let host = req.config_str("host").ok_or("config.host is required")?.to_string();
        let (rack, slot) = resolve_rack_slot(&req.config)?;
        let dbs = {
            let client = self.client_for(&host, rack, slot)?;
            s7::blocks::list_data_blocks(client)
        };
        let dbs = match dbs {
            Ok(d) => d,
            Err(e) => { self.reset(); return Err(format!("s7 discover: {e}")); }
        };
        let nodes = dbs
            .into_iter()
            .map(|b| {
                let label = match b.size {
                    Some(sz) => format!("DB{} ({} bytes)", b.number, sz),
                    None => format!("DB{}", b.number),
                };
                let mut n = Node::leaf(format!("DB{}", b.number), label, "db");
                n.meta = serde_json::json!({ "db": b.number, "size_bytes": b.size });
                n
            })
            .collect();
        Ok(Discovery { schema_kind: "s7_blocks".into(), nodes })
    }
}

/// Read one scalar described by `sel` from a connected client.
fn read_value(client: &mut S7Client, sel: &Selection) -> Result<f64, String> {
    if sel.dtype == DType::Bool {
        let b = client
            .read_bit(sel.area, sel.db, sel.byte, sel.bit)
            .map_err(|e| e.to_string())?;
        return Ok(if b { 1.0 } else { 0.0 });
    }
    let len = sel.dtype.byte_len();
    let mut buf = vec![0u8; len];
    client
        .read_area(sel.area, sel.db, sel.byte, S7_WL_BYTE, &mut buf)
        .map_err(|e| e.to_string())?;
    sel.dtype.decode(&buf)
}

/// Turn a rust7 S7Error string into an actionable message for the common cases.
fn map_s7_err(e: &str, sel: &Selection) -> String {
    if e.contains("Invalid address") {
        format!("s7 read failed: {e} — address out of range, or the DB is optimized \
                 (switch DB{} to standard/non-optimized access in TIA Portal)", sel.db)
    } else if e.contains("not found") || e.contains("Resource not found") {
        format!("s7 read failed: {e} — DB{} does not exist in the CPU", sel.db)
    } else {
        format!("s7 read failed: {e}")
    }
}

fn main() {
    run(S7Connector::new());
}
```

- [ ] **Step 1 (client.rs accessor — LYMON DELTA):** add inside `impl S7Client`:

```rust
    // LYMON DELTA: low-level request/response for the discover extension (blocks.rs).
    // Sends a full telegram, reads the TPKT+ISO header (read_exact), validates it, then
    // reads the S7 payload (read_exact). Returns the S7 payload bytes (without the 7-byte
    // TPKT/ISO header). Connection-management (reconnect on error) stays the caller's job.
    pub(crate) fn exchange(&mut self, request: &[u8]) -> Result<Vec<u8>, S7Error> {
        if !self.connected { return Err(S7Error::NotConnected); }
        let stream = self.stream.as_mut().unwrap();
        stream.write_all(request)?;
        let mut iso = [0u8; TPKT_ISO_LEN];
        stream.read_exact(&mut iso)?;
        let s7_size = check_iso_packet(self.pdu_length, &mut iso)?;
        let mut payload = vec![0u8; s7_size];
        stream.read_exact(&mut payload)?;
        Ok(payload)
    }
    /// Negotiated PDU length (for chunking userdata responses if needed).
    pub(crate) fn pdu_len(&self) -> u16 { self.pdu_length }
```

- [ ] **Step 2 (blocks.rs):** implement `list_data_blocks` by porting snap7's
  *List Blocks of Type* userdata telegram (block type `DB`). The S7comm userdata request is:
  TPKT(4) + COTP(3) + S7-userdata-header(10: `32 07 00 00 <pduref:2> <parlen:2> <datlen:2>`) +
  param(8: `00 01 12 04 11 43 02 00`, where `43`=block functions group request, `02`=list blocks
  of type) + data(`FF 09 00 04 30 30 41 41` = retval FF, ts OCTET-STRING len 4, ASCII `"00"` +
  block type `"A"`(DB)`A`...). The exact byte sequence is taken from snap7's `Cli_ListBlocksOfType`
  and validated against a Wireshark trace per the RUNBOOK. The response carries N×4-byte entries
  `<block_number:2> <flags:1> <lang:1>`; we extract the block numbers.

```rust
//! Discover extension: enumerate the CPU's data blocks via the S7 "list blocks of
//! type" userdata service. Ported from snap7's Cli_ListBlocksOfType (block type DB).
//! Coarse by design — S7comm exposes DB numbers/sizes, not field layouts.

use super::client::{S7Client, S7Error};

pub struct BlockRef {
    pub number: u16,
    pub size: Option<u32>,
}

/// "List blocks of type = DB" userdata telegram (fixed; ref: snap7 Cli_ListBlocksOfType).
/// NOTE: exact bytes confirmed against snap7 source + Wireshark during bring-up (RUNBOOK).
const LIST_DB_TELEGRAM: &[u8] = &[
    0x03, 0x00, 0x00, 0x1d,             // TPKT, total len 29
    0x02, 0xf0, 0x80,                   // COTP
    0x32, 0x07, 0x00, 0x00, 0x00, 0x00, // S7 userdata hdr: proto, ROSCTR=7, redundancy, pduref
    0x00, 0x08,                         // parameter length = 8
    0x00, 0x04,                         // data length = 4
    0x00, 0x01, 0x12, 0x04, 0x11, 0x43, 0x02, 0x00, // param: list blocks of type (43/02)
    0x00, 0x0a, 0x00, 0x00,             // data: retval/transport/len placeholder + block type
];

/// Enumerate DBs present in the CPU.
pub fn list_data_blocks(client: &mut S7Client) -> Result<Vec<BlockRef>, S7Error> {
    let payload = client.exchange(LIST_DB_TELEGRAM)?;
    Ok(parse_block_list(&payload))
}

/// Parse the userdata response payload into block numbers. The data section is a
/// sequence of 4-byte records: block_number(BE u16), flags(u8), lang(u8). The data
/// starts after the userdata param+data headers; we locate it by the 0xFF return code.
fn parse_block_list(payload: &[u8]) -> Vec<BlockRef> {
    // Find the userdata data header return-code 0xFF, then OCTET STRING (09) + length.
    let mut out = Vec::new();
    if let Some(pos) = find_data_start(payload) {
        let mut i = pos;
        while i + 4 <= payload.len() {
            let number = u16::from_be_bytes([payload[i], payload[i + 1]]);
            out.push(BlockRef { number, size: None });
            i += 4;
        }
    }
    out
}

/// Locate the start of the block-record list inside the userdata response: after the
/// param (offset ~12) the data section begins with retcode 0xFF, transport 0x09, len(2).
fn find_data_start(payload: &[u8]) -> Option<usize> {
    // userdata response param is 12 bytes; data header is retcode(1)+transport(1)+len(2).
    // record list begins at param_end + 4.
    const PARAM_LEN: usize = 12;
    if payload.len() >= PARAM_LEN + 4 && payload[PARAM_LEN] == 0xFF {
        Some(PARAM_LEN + 4)
    } else {
        None
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn parses_block_records() {
        // param(12) + data hdr [FF 09 00 08] + DB1,DB17 records (num,flags,lang)
        let mut p = vec![0u8; 12];
        p.extend_from_slice(&[0xFF, 0x09, 0x00, 0x08]);
        p.extend_from_slice(&[0x00, 0x01, 0x00, 0x00]); // DB1
        p.extend_from_slice(&[0x00, 0x11, 0x00, 0x00]); // DB17
        let blocks = parse_block_list(&p);
        assert_eq!(blocks.iter().map(|b| b.number).collect::<Vec<_>>(), vec![1, 17]);
    }
}
```
